27void Blocking( // Simple Blocking
28 int *cind, int *rptr,
29 int **b_cind, int **b_rptr, int **block,
30 int N, int bsize)
31{
32 *block = utils::SafeMalloc<int>(N);
33 for(int i=0; i<N; i++) { (*block)[i] = i; }
34
35 *b_rptr = utils::SafeMalloc<int>(N/bsize+1);
36 *b_cind = utils::SafeMalloc<int>(rptr[N]); // At most rptr[N]
37 int cnt = 0;
38 (*b_rptr)[0] = 0;
39 int *ptr = utils::SafeMalloc<int>(bsize);
40 for(int i=0; i<N; i+=bsize) {
41 // Initialize "ptr"
42 for(int j=0; j<bsize; j++) { ptr[j] = rptr[(*block)[i+j]]; }
43 while(true) {
44 // Find minimum col index
45 int min = N;
46 for(int j=0; j<bsize; j++) {
47 if(ptr[j] != -1 && cind[ptr[j]] < min) {
48 min = cind[ptr[j]];
49 }
50 }
51 if(min == N) break;
52 (*b_cind)[cnt] = min/bsize;
53 for(int j=0; j<bsize; j++) {
54 if(ptr[j] == -1) continue;
55 while(cind[ptr[j]]/bsize == min/bsize) {
56 ptr[j]++;
57 if(ptr[j] >= rptr[(*block)[i+j]+1]) {
58 ptr[j] = -1; break;
59 }
60 }
61 }
62 cnt++;
63 }
64 (*b_rptr)[i/bsize+1] = cnt;
65 }

76void GetAdjacency(
77 int *cind, int *rptr,

81 if(isSym) {
82 *am_cind = utils::SafeMalloc<int>(rptr[N]);

84 utils::Copy<int>(cind, *am_cind, rptr[N]);

90 for(int i=0; i<N; i++) {

92 len[cind[j]]++;

97 cptr[i+1] = cptr[i] + len[i];

100 for(int i=0; i<N; i++) {
101 for(int j=rptr[i]; j<rptr[i+1]; j++) {
102 rind[cptr[cind[j]]+len[cind[j]]] = i;
103 len[cind[j]]++;
104 }
105 }
106 *am_cind = utils::SafeMalloc<int>(rptr[N]*2);
107 *am_rptr = utils::SafeMalloc<int>(N+1);
108 (*am_rptr)[0] = 0;
109 int cnt = 0;
110 for(int i=0; i<N; i++) {
111 int csr_ptr = rptr[i];
112 int csc_ptr = cptr[i];
113 while(csr_ptr < rptr[i+1] || csc_ptr < cptr[i+1]) {
114 int csr_idx = (csr_ptr < rptr[i+1])? cind[csr_ptr] : N;
115 int csc_idx = (csc_ptr < cptr[i+1])? rind[csc_ptr] : N;
116 if(csr_idx < csc_idx) {
117 (*am_cind)[cnt] = csr_idx;
118 csr_ptr++;
119 }else if(csr_idx == csc_idx){
120 (*am_cind)[cnt] = csr_idx;
121 csr_ptr++;
122 csc_ptr++;
123 }else {
124 (*am_cind)[cnt] = csc_idx;
125 csc_ptr++;
126 }
127 cnt++;
128 }
129 (*am_rptr)[i+1] = cnt;
130 }
131 free(rind);
132 free(cptr);
133 free(len);

144void Coloring( // Greedy Coloring
145 int *cind, int *rptr, int **color, int *num_color, int N)
146{
147 *color = utils::SafeCalloc<int>(N);
148 int *visit = utils::SafeCalloc<int>(N);
149 int now = 1;
150 int cnt = 0;
151 while(cnt < N) {
152 for(int i=0; i<N; i++) {
153 if(visit[i] == now || (*color)[i] != 0) continue;
154 (*color)[i] = now;
155 cnt++;
156 for(int j=rptr[i]; j<rptr[i+1]; j++) {
157 visit[cind[j]] = now;
158 }
159 }
160 now++;
161 }
162 *num_color = now - 1;
163 free(visit);
164}
