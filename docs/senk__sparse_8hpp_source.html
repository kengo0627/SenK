69template <typename T> inline
70void SpmvSell(T *val, int *cind, int *wid, int len, T *x, T *y, int N)
71{
72 int block = (N+len-1)/len;
73 #pragma omp parallel for
74 for(int i=0; i<block; i++) {
75 int start = wid[i] * len;
76 int temp = (i==len-1 && N%len!=0) ? N % len : len;
77 for(int k=0; k<temp; k++) {
78 y[i*len+k] = val[start+k] * x[cind[start+k]];
79 }
80 for(int j=1; j<wid[i+1]-wid[i]; j++) {
81 int off = start+j*len;
82 for(int k=0; k<temp; k++) {
83 y[i*len+k] += val[off+k] * x[cind[off+k]];
84 }
85 }

123 // L is assumed to be general upper triangular.

127 int j;

183template <typename T, int bnl, int bnw> inline
184void SptrsvBcsr_u(
185 T *bval, int *bcind, int *brptr,
186 T *x, T *y, int N)
187{
188 int b_size = bnl * bnw;
189 int b_rem = bnl / bnw;
190 for(int i=N-bnl; i>=0; i-=bnl) {
191 int bidx = i / bnl;
192 #pragma omp simd simdlen(bnl)
193 for(int j=0; j<bnl; j++) {
194 y[i+j] = x[i+j];
195 }
196 for(int j=brptr[bidx+1]-1; j>=brptr[bidx]+b_rem; j--) {
197 int x_ind = bcind[j]*bnw;
198 for(int l=0; l<bnw; l++) {
199 int off = j*b_size+l*bnl;
200 #pragma omp simd simdlen(bnl)
201 for(int k=0; k<bnl; k++) {
202 y[i+k] -= bval[off+k] * y[x_ind+l];
203 }
204 }
205 }
206 int pos = brptr[bidx]+b_rem-1;
207 for(int k=b_rem-1; k>=0; k--) {
208 for(int j=bnw-1; j>=0; j--) {
209 int off = pos*b_size+j*bnl;
210 int idx = k*bnw+j;
211 y[i+idx] *= bval[off+idx];
212 for(int l=k*bnw+j-1; l>=0; l--) {
213 y[i+l] -= bval[off+l] * y[i+idx];
214 }
215 }
216 pos--;
217 }
