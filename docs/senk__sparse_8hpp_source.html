69template <typename T> inline
70void SpmvSell(T *val, int *cind, int *wid, int len, T *x, T *y, int N)
71{
72 int block = (N+len-1)/len;
73 #pragma omp parallel for
74 for(int i=0; i<block; i++) {
75 int start = wid[i] * len;
76 int temp = (i==len-1 && N%len!=0) ? N % len : len;
77 for(int k=0; k<temp; k++) {
78 y[i*len+k] = val[start+k] * x[cind[start+k]];
79 }
80 for(int j=1; j<wid[i+1]-wid[i]; j++) {
81 int off = start+j*len;
82 for(int k=0; k<temp; k++) {
83 y[i*len+k] += val[off+k] * x[cind[off+k]];
84 }
85 }

218 T *x, T *y, int N)
219{
220 // L is assumed to be unit lower triangular.
221 int b_size = bnl * bnw;
222 for(int i=0; i<N; i+=bnl) {
223 int bidx = i / bnl;
224 #pragma omp simd simdlen(bnl)
225 for(int j=0; j<bnl; j++) {
226 y[i+j] = x[i+j];
227 }
228 for(int j=brptr[bidx]; j<brptr[bidx+1]; j++) {
229 int x_ind = bcind[j]*bnw;

232 #pragma omp simd simdlen(bnl)
233 for(int k=0; k<bnl; k++) {
234 y[i+k] -= bval[off+k] * y[x_ind+l];

253void SptrsvBcsr_u(
254 T *bval, int *bcind, int *brptr,
255 T *x, T *y, int N)
256{
257 int b_size = bnl * bnw;
258 int b_rem = bnl / bnw;
259 for(int i=N-bnl; i>=0; i-=bnl) {
260 int bidx = i / bnl;
261 #pragma omp simd simdlen(bnl)
262 for(int j=0; j<bnl; j++) {
263 y[i+j] = x[i+j];
264 }
265 for(int j=brptr[bidx+1]-1; j>=brptr[bidx]+b_rem; j--) {
266 int x_ind = bcind[j]*bnw;
267 for(int l=0; l<bnw; l++) {
268 int off = j*b_size+l*bnl;
269 #pragma omp simd simdlen(bnl)
270 for(int k=0; k<bnl; k++) {
271 y[i+k] -= bval[off+k] * y[x_ind+l];
272 }
273 }
274 }
275 int pos = brptr[bidx]+b_rem-1;
276 for(int k=b_rem-1; k>=0; k--) {
277 for(int j=bnw-1; j>=0; j--) {
278 int off = pos*b_size+j*bnl;
279 int idx = k*bnw+j;
280 y[i+idx] *= bval[off+idx];
281 for(int l=k*bnw+j-1; l>=0; l--) {
282 y[i+l] -= bval[off+l] * y[i+idx];
283 }
284 }
285 pos--;
286 }
